#include <stdio.h>

int main()
{
    int num;
    
    printf("숫자를 입력하세요.");
    scanf("%d",&num);
    
    //조건문
    if (num % 2 == 0)	// if = 조건 문장 (소괄호 필요)
    {
        printf("짝수이다.\n"); 		//if조건이 참이면 실행되는 블록
        printf("2로 나눈 나머지가 0\n");
    }										//>:초과  <:미만 >=:이상, <=:이하, ==:같다, !=:같지 않다
    else
    {
        printf("홀수이다.\n");		//위의 if 조건이 거짓일 때 실행되는 블록
        printf("2로 나눈 나머지가 1\n");
    }
    
    return 0;
}
