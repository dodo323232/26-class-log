// 문제) 1글자를 받아 대문자, 소문자, 숫자로 분류
#include <stdio.h>

int main(void)
{
	char ch; //문자1글자 변수					//컴퓨터는 문자를 숫자로 기억한다		%d는 4바이트 %f는 8바이트 %c는 1바이트
	printf("문자를 입력하세요: ");
	//scanf_s("%c", &ch);
	ch = getchar(); //한글자만 입력 받을때 getchar를 사용
	if (ch >= 'A' && ch <= 'Z')		//65 ~ 90 대문자 코드
		//(ch >= 65 && ch <= 90)
		printf("대문자 입니다.");
	else if (ch >= 'a' && ch <= 'z')	//97 ~ 122 소문자 코드
		//(ch >= 97 && ch <= 122										
		printf("소문자 입니다.");
	else if (ch >= '0' && ch <= '9')	// 48 ~ 57 숫자 코드
		//(ch >= 48 && ch <= 57)
		printf("숫자 입니다.");
	else
		printf("기타 문자");
	
	
	return 0;
}
